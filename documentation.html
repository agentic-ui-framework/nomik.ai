Your First Payment

Step-by-step walkthrough to execute your first autonomous payment with detailed explanations of each component and what happens behind the scenes.

1

Set Up Your Environment

Create a new project and install the Nomik SDK:

mkdir my-nomik-project
cd my-nomik-project
npm init -y
npm install @nomik/sdk dotenv

Create a .env file with your API credentials:

NOMIK_API_KEY=your_sandbox_api_key_here
NOMIK_ENVIRONMENT=sandbox
2

Initialize the Client

Create first-payment.js and set up the Nomik client:

require('dotenv').config();
const { NomikClient } = require('@nomik/sdk');

const nomik = new NomikClient({
  apiKey: process.env.NOMIK_API_KEY,
  environment: process.env.NOMIK_ENVIRONMENT,
  debug: true // Enable detailed logging for learning
});

console.log('Nomik client initialized successfully');
3

Create an Agent Identity

Every payment needs an agent with proper authorization:

async function createAgent() {
  try {
    const agent = await nomik.agents.create({
      name: 'tutorial_payment_agent',
      entity_name: 'Tutorial Company',
      entity_jurisdiction: 'Netherlands',
      authorized_by: 'tutorial@example.com',
      capabilities: ['payment_processing'],
      max_delegation_amount: 50000
    });

    console.log('Agent created:', agent);
    return agent.id;
  } catch (error) {
    console.error('Failed to create agent:', error.message);
    throw error;
  }
}
4

Execute Your First Payment

Now execute a test payment with the sandbox:

async function executeFirstPayment(agentId) {
  try {
    console.log('Executing payment...');

    const payment = await nomik.payments.create({
      agent_id: agentId,
      amount: 1000.00, // ‚Ç¨10.00 test payment
      currency: 'EUR',
      to: 'sandbox_recipient_001', // Sandbox test recipient
      description: 'My first autonomous payment',
      metadata: {
        tutorial: 'first_payment',
        timestamp: new Date().toISOString()
      }
    });

    console.log('Payment executed successfully!');
    console.log('Payment ID:', payment.id);
    console.log('Status:', payment.status);
    console.log('Settlement time:', payment.settlement_time);

    return payment;
  } catch (error) {
    console.error('Payment failed:', error.message);
    if (error.details) {
      console.error('Error details:', error.details);
    }
    throw error;
  }
}

// Run the complete tutorial

Authentication

Nomik uses API keys and cryptographic agent identities for secure authentication. All requests must include proper authentication headers.

API Key Authentication

Use your API key for server-to-server authentication:

curl -H "Authorization: Bearer your_api_key_here" \
     -H "Content-Type: application/json" \
     https://api.nomik.ai/v1/agents

Agent Identity Verification

Each agent must have a cryptographically verified identity:

from nomik import Agent, Identity

# Create cryptographic identity
identity = Identity.generate(
    entity_name="YourCompany",
    entity_type="corporation",
    jurisdiction="EU"
)

# Agent inherits identity
agent = Agent(
    name="procurement_agent",
    identity=identity,
    api_key="your_api_key"
)

Security Best Practices

  • Never expose API keys in client-side code
  • Rotate API keys regularly (recommended: 90 days)
  • Use environment variables for key storage
  • Implement proper key management in production

TypeScript Definitions

Full TypeScript support with comprehensive type definitions:

import { NomikClient, PaymentRequest, Agent, Policy } from '@nomik/sdk';

interface PaymentHandler {
  client: NomikClient;
  defaultAgent: Agent;
  policy: Policy;
}

class AutonomousPaymentService implements PaymentHandler {
  client: NomikClient;
  defaultAgent: Agent;
  policy: Policy;

  constructor(config: NomikConfig) {
    this.client = new NomikClient(config);
  }

  async processInvoicePayment(
    invoice: Invoice,
    approval?: ApprovalToken
  ): Promise {
    const request: PaymentRequest = {
      agentId: this.defaultAgent.id,
      amount: invoice.totalAmount,
      currency: invoice.currency,
      to: invoice.supplier.paymentDetails,
      metadata: {
        invoiceId: invoice.id,
        dueDate: invoice.dueDate.toISOString(),
        approvalChain: approval?.chain
      }
    };

    // Policy check before execution
    const policyCheck = await this.policy.evaluate(request);
    if (!policyCheck.approved) {
      throw new PolicyViolationError(policyCheck.reason);
    }

    return await this.client.payments.execute(request);
  }
}

Batch Processing

Efficient bulk payment processing with error handling

// Batch payment processing pattern
class BatchPaymentProcessor {
  constructor(nomikClient) {
    this.nomik = nomikClient;
    this.batchSize = 50;
    this.maxRetries = 3;
  }

  async processBatch(payments) {
    const batches = this.chunkArray(payments, this.batchSize);
    const results = [];

    for (const batch of batches) {
      const batchResults = await this.processBatchChunk(batch);
      results.push(...batchResults);
    }

    return this.generateReport(results);
  }

  async processBatchChunk(batch) {
    const promises = batch.map(payment =>
      this.processWithRetry(payment)
    );

    return await Promise.allSettled(promises);
  }

  async processWithRetry(payment, retryCount = 0) {
    try {
      return await this.nomik.payments.create(payment);
    } catch (error) {
      if (retryCount < this.maxRetries && this.isRetryable(error)) {
        await this.delay(Math.pow(2, retryCount) * 1000);
        return this.processWithRetry(payment, retryCount + 1);
      }
      throw error;
    }
  }
}

// Microservice integration pattern (Go)
package payments

import (
    "context"
    "github.com/nomik-ai/go-sdk"
)

type PaymentService struct {
    nomik       *nomik.Client
    policySvc   PolicyService
    auditSvc    AuditService
}

func (s *PaymentService) ProcessPayment(ctx context.Context, req *PaymentRequest) (*Payment, error) {
    // Policy check
    if err := s.policySvc.ValidatePayment(ctx, req); err != nil {
        return nil, err
    }

    // Execute payment
    payment, err := s.nomik.Payments.Create(ctx, req)
    if err != nil {
        return nil, err
    }

    // Audit trail
    go s.auditSvc.LogPayment(payment)

    return payment, nil
}
